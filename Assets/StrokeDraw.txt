using System.Collections;
using UnityEngine;

public class StrokeDraw : MonoBehaviour
{
    public float duration = 0.5f;

    public enum Axis { X, Y, Z }
    public Axis growAxis = Axis.X;

    public enum PivotSide { Start, End }
    public PivotSide pivotSide = PivotSide.Start;

    public AnimationCurve ease = AnimationCurve.EaseInOut(0, 0, 1, 1);

    private Vector3 originalScale;
    private Vector3 originalPos;
    private bool initialized = false;

    void Init()
    {
        if (initialized) return;
        originalScale = transform.localScale;
        originalPos = transform.localPosition;
        initialized = true;
    }

    void Awake()
    {
        Init();
        transform.localScale = Vector3.zero;
    }

    void OnEnable()
    {
        Init();
        transform.localScale = Vector3.zero;
    }

    public void PlayStroke()
    {
        Init();
        StopAllCoroutines();
        StartCoroutine(Animate());
    }

    IEnumerator Animate()
    {
        float t = 0;
        while (t < 1f)
        {
            t += Time.deltaTime / duration;
            float eased = ease.Evaluate(Mathf.Clamp01(t));

            Vector3 s = originalScale;
            Vector3 pos = originalPos;

            switch (growAxis)
            {
                case Axis.X:
                    s.x = Mathf.Lerp(0.001f, originalScale.x, eased);
                    if (pivotSide == PivotSide.Start)
                        pos.x = originalPos.x - (originalScale.x - s.x) / 2f;
                    else
                        pos.x = originalPos.x + (originalScale.x - s.x) / 2f;
                    break;

                case Axis.Y:
                    s.y = Mathf.Lerp(0.001f, originalScale.y, eased);
                    if (pivotSide == PivotSide.Start)
                        pos.y = originalPos.y + (originalScale.y - s.y) / 2f;
                    else
                        pos.y = originalPos.y - (originalScale.y - s.y) / 2f;
                    break;
            }

            transform.localScale = s;
            transform.localPosition = pos;
            yield return null;
        }

        transform.localScale = originalScale;
        transform.localPosition = originalPos;
    }
}
